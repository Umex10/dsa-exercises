/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public boolean hasCycle(ListNode head) {

        ListNode slow = head;
        ListNode fast = head;

        // [3,2,0,-4]
        // slow --> 2 | fast --> 0
        // slow --> 0 | fast --> 2
        // slow --> -4 | fast --> -4
        // found

        while (fast != null && fast.next != null) {

            slow = slow.next; // + 1 node
            fast = fast.next.next; // + 2 nodes

            if (slow == fast) {
                return true;
            }

        }

        return false;

    }
}
