/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public boolean hasCycle(ListNode head) {

        ListNode dummy = new ListNode(0);
        ListNode slow = dummy;
        ListNode fast = head;

        while (fast != null && fast.next != null) {

            if (slow == fast) {
                return true;
            }

            slow = head;
            head = head.next;
            fast = fast.next.next;

        }

        return false;        
    }

}
