class Solution {
    public int lengthOfLastWord(String s) {

        int length = 0;

        for (int i = s.length() - 1; i >= 0; i--) {

            if (s.charAt(i) != ' ') {
                for (int u = i; u >= 0; u--) {
                    if (s.charAt(u) == ' ') {
                        return length;
                    }
                    length++;

                }
                return length;
            }
        }

        return length;

    }
}
